class MetaTable(type):
	def __getattr__(cls, key):
		temp = key.split("__")
		name = temp[0]
		alias = None
		if len(temp) > 1:
			alias = temp[1]
		return cls(name, alias)

class Table(object, metaclass=MetaTable):
	def __init__(self, name, alias=None):
		self._name = name
		self._alias = alias
